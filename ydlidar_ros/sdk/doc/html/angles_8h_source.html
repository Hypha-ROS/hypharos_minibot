114 static inline double two_pi_complement(double angle) {
115  //check input conditions
116  if (angle > 2 * M_PI || angle < -2.0 * M_PI) {
117  angle = fmod(angle, 2.0 * M_PI);
118  }
119 
120  if (angle < 0) {
121  return (2 * M_PI + angle);
122  } else if (angle > 0) {
123  return (-2 * M_PI + angle);
124  }
125 
126  return (2 * M_PI);
127 }
